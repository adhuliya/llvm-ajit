Value types used in llvm backend.

        constexpr MVT() = default;
        constexpr MVT(SimpleValueType SVT) : SimpleTy(SVT) {}

        bool operator>(const MVT& S)  const { return SimpleTy >  S.SimpleTy; }
        bool operator<(const MVT& S)  const { return SimpleTy <  S.SimpleTy; }
        bool operator==(const MVT& S) const { return SimpleTy == S.SimpleTy; }
        bool operator!=(const MVT& S) const { return SimpleTy != S.SimpleTy; }
        bool operator>=(const MVT& S) const { return SimpleTy >= S.SimpleTy; }
        bool operator<=(const MVT& S) const { return SimpleTy <= S.SimpleTy; }
